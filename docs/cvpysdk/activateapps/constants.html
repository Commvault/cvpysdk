TrainingStatus                  -       Enum class for classifier training status

"""
import copy
from enum import Enum


class TrainingStatus(Enum):
    """Class to maintain training status for classifier"""
    NOT_APPLICABLE = 0
    CREATED = 1

    FAILED = 3
    COMPLETED = 4
    CANCELLED = 5
    NOT_USABLE = 6
